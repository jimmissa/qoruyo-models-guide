ܙܟ݂ܵܝܗܝ ܗ̄ܘ̣ܵܐ ܠܫܲܝܢܵܐ ܟܢܘܼܫܝܵܐ ܪܲܒܵܐ ܕܲܚܕ݂ܝܼܪ ܗ̄ܘܵܐ ܠܹܗ:
ܘܲܡܠܵܐ ܠܚܘܼܪܒܵܐ ܤܡܵܟܹ̈ܐ ܕܠܵܐ ܤܵܟ݂ ܕܐܲܓ݂ܢܝܼ ܗ̄ܘܵܐ ܒܹܗ܀
ܡܸܟܵܐ ܘܡܸܟܵܐ ܟܠ ܢܸܨܚܵܢܵܘ̈ܗܝ ܡܸܫܬܲܡܥܝܼܢ ܗ̄ܘ̣ܵܘܐ
ܘܟ݂ܠ ܟܹܐܒ݂ܬܵܢܹ̈ܐ ܒܗܵܠܹܝܢ ܛܸܒܹ̈ܐ ܡܸܤܬܲܡܟ݁ܝܼܢ ܗ̄ܘܵܘ܀
ܐܲܪܓܸܫ̣ ܗ̄ܘ̣ܵܐ ܒܹܗ ܓܲܒ݂ܪܵܐ ܛܪܝܼܕ݂ܵܐ (ܘܲܒ݂ܚܘܼܠܡܵܢܵܘ̈ܗܝ:‏
ܘܲܚ̣ܒܲܨ ܐܵܦ ܗ݀ܘ ܢܸܥܘܿܠ ܢܸܤܦܸܐ ܥܘܼܬ݂ܪܵܐ ܡܸܢܹܗ܀
ܠܟܸܢܫܵܐ ܢܸܥܘܿܠ ܡܸܤܬܲܪܲܕ݂ ܗ̄ܘ̣ܵܐ ܡ̣ܢ ܢܵܡܘܿܤܵܐ:
ܕܛܲܡ̣ܐܹܗ ܘܲܫܕܵܝܗܝ ܐܲܝܟ݂ ܢܘܼܟ݂ܪܵܝܵܐ ܠܒܲܪ ܡ̣ܢ ܥܲܡܵܐ܀
ܕܢܸܓ݂ܠܸܐ ܦܘܼܡܹܗ ܘܲܢܗܲܠܸܟ݂ ܗ̄ܘܵܐ ܠܵܐ ܡܸܫܟܲܚ ܗ̄ܘ̣ܵܐ:
ܕܡܘܼܫܹ̈ܐ ܦܲܩܲܕܹܗ ܕܢܸܗܘܸܐ ܡܟܲܤ̣ܲܝ ܥܲܠ ܤܸܦܵܘܵܬܹܗ܀
ܫܸܡܥܹܗ ܕܝܼܫܘܿܥ ܡܲܪܗܸܒ݂ ܗ̄ܘ̣ܵܐ ܠܹܗ ܕܢܹܐܬܸܐ ܨܹܐܕܵܘܗܝ:
ܘܲܕܢܸܥܘܿܠ ܗ̄ܘ̣ܵܐ ܘܢܸܚܙܹܝܘܗܝ ܥܲܡܵܐ ܡܸܤܬܲܪܲܕ݂ ܗ̄ܘܵܐ܀
ܐܵܬܹܝܢ ܗ̄ܘ̣ܵܘ ܠܹܗ ܛܸܒܹܐ ܕܬܲܗܪܵܐ ܡ̣ܢ ܟܠ ܓܲܒ̈ܝܼܢ:
ܘܲܙܪܝܼܒ݂ ܗ̄ܘ̣ܵܐ ܗ݀ܘ ܕܲܠܓ݂ܵܘ ܥܲܡܵܐ ܠܵܐ ܢܸܥܘܿܠ ܗ̄ܘܵܐ܀
ܡܘܼܫܹܐ ܓܙܲܡ݂ ܠܹܗ ܕܢܸܗܘܸܐ ܡܢܲܟ݂ܪܲܝ ܡ̣ܢ ܝܘܼܪܬܵܢܹܗ:
ܘܠܵܐ ܝܲܗ̣̄ܒ݂ ܗ̄ܘ̣ܵܐ ܠܹܗ ܐܲܬ݂ܪܵܐ ܒܥܲܟܵܐ ܠܲܡܗܲܠܵܟ݂ܘܼ܀
ܘܟܲܕ݂ ܕܵܚܹܠ ܗ̄ܘܵܐ ܡ̣ܢ ܢܵܡܘܿܤܵܐ ܕܐܲܪܚܸܩ̣ ܘܲܫ̣ܕܵܝܗܝ:
ܫܸܡܥܹܐ ܕܬܲܗܪܵܐ ܢܵܬ݂ܦ̇ܝܼܢ ܗ̄ܘ̣ܵܘ ܠܹܗ ܠܘܵܬ݂ ܦܵܪܘܿܩܲܢ܀
ܐ̄ܬܵܘ ܠܹܗ ܛܸܒܹ̈ܐ ܕܝܼܫܘܿܥ ܐܵܤܝܵܐ ܩܵܡ̣ ܡ̣ܢ ܥܲܡܵܐ:‏
ܘܡܲܐܤܸܐ ܟܪ̈ܝܼܗܹܐ ܘܡܲܦܸܩ ܕܲܝܘܹ̈ܐ ܘܪܵܕܹܦ ܫܹܐܕܹ̈ܐ܀
